---para cambiar dinámicamente entre los estilos normales y los de alto contraste---

const botonAltoContraste = document.getElementById('alto-contraste');

botonAltoContraste.addEventListener('click', () => {
    document.body.classList.toggle('alto-contraste');

    if (document.body.classList.contains('alto-contraste')) {
        botonAltoContraste.textContent = 'Modo Normal';
        botonAltoContraste.setAttribute('aria-label', 'Desactivar modo de alto contraste');
    } else {
        botonAltoContraste.textContent = 'Alto Contraste';
        botonAltoContraste.setAttribute('aria-label', 'Activar modo de alto contraste');
    }
});


---Esta API se utiliza para lectura---

// Seleccionar el botón de lectura en voz alta
const botonLeerVozAlta = document.getElementById('leer-voz-alta');

// Función para leer el contenido en voz alta
function leerEnVozAlta() {
    const textoALeer = document.querySelector('main').innerText;
    const utterance = new SpeechSynthesisUtterance(textoALeer);
    utterance.lang = 'es-ES';

    speechSynthesis.speak(utterance);

    botonLeerVozAlta.textContent = 'Detener Lectura';
    botonLeerVozAlta.setAttribute('aria-label', 'Detener lectura en voz alta');

    botonLeerVozAlta.onclick = () => {
        if (speechSynthesis.speaking) {
            speechSynthesis.cancel();
            botonLeerVozAlta.textContent = 'Leer en Voz Alta';
            botonLeerVozAlta.setAttribute('aria-label', 'Leer contenido en voz alta');
        }
    };
}

// Asignar la función al botón
botonLeerVozAlta.addEventListener('click', leerEnVozAlta);
